class Solution {
    public int[] twoSum(int[] nums, int target) {
        if(nums.length == 1){
            return new int[0];
        }
        int i = 0;
        int j = nums.length-1;
        while(i < j){
            if(nums[i] + nums[j] == target) break;
            if(nums[i] + nums[j] > target){
                j --;
            }else{
                i ++;
            }   
        }
        int[] res = {nums[i], nums[j]};
        return res;
    }
}
